package com.backjoon.back9363;

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.HashMap;
import java.util.Map;
import java.util.StringTokenizer;

public class Main {
    static final int MAX = 1000000;
    static int[] spf = new int[MAX + 1];
    public static BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
    public  static void main(String[] args) throws IOException {
        int T = Integer.parseInt(br.readLine());

        buildSPF();

        for (int i = 0; i < T; i++) {
            StringTokenizer st = new StringTokenizer(br.readLine());
            HashMap<Integer, Integer> countMap = new HashMap<>();
            int N = Integer.parseInt(st.nextToken());
            int M = Integer.parseInt(st.nextToken());

            st = new StringTokenizer(br.readLine());
            for (int j = 0; j < N; j++) {
                int a = Integer.parseInt(st.nextToken());
                addFactor(a,1,countMap);
            }
            st = new StringTokenizer(br.readLine());
            for (int j = 0; j < M; j++) {
                int b = Integer.parseInt(st.nextToken());
                addFactor(b,-1,countMap);
            }

            int A1 = 1;
            int B1 = 1;
            // 남은 개수 계산
            for(int key : countMap.keySet()){
                int count = countMap.get(key);

                if(count > 0) {
                    A1 *= key * count;
                } else if(count < 0) {
                    B1 *= key * -count;
                }
            }

            System.out.print("Case #" + (i + 1) + ": ");
            System.out.println(A1 + " / " + B1);

        }
    }

    // 해당 소인수 개수 파악 A이면 +, B이면 -
    static void addFactor(int x, int d, Map<Integer, Integer> countMap) {
        while(x > 1) {
            int p = spf[x];
            int cnt = 0;
            while(x%p == 0) {
                x/=p;
                cnt++;
            }

            countMap.put(p, countMap.getOrDefault(p, 0) + d * cnt);
        }
    }

    // 최소가 되는 소인수 조사
    static void buildSPF() {
        for (int i = 1; i <= MAX; i++) spf[i] = i;

        for (int i = 2; i * i <= MAX; i++) {
            if (spf[i] == i) { // prime
                for (int j = i * i; j <= MAX; j += i) {
                    if (spf[j] == j) spf[j] = i;
                }
            }
        }
    }
}
