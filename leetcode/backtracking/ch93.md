## leetcode ch93. Restore IP Addresses

```java

class Solution {
    List<String> ans = new ArrayList<>();

    public List<String> restoreIpAddresses(String s) {
        splitIp(s,"",0);

        return ans;
    }

    public void splitIp(String s, String ip, int count) {
        if(count > 4) return;
        if(count == 4 && s.length() == 0) {
            ans.add(ip);
            return;
        }

        for(int i = 1; i <= 3; i++) {
            if(s.length() < i) break;
            String num = s.substring(0,i);
            if(num.charAt(0) == '0' && num.length() != 1) break;
            else if (Integer.parseInt(num) <= 255) {
                String next = ip;
                if(count == 0) next = ip + num;
                else next = ip + "." + num;
                splitIp(s.substring(i,s.length()), next, count + 1);
            }
        }
    }

    
}
```
