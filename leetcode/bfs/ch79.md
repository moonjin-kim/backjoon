## 79. Word Search

```java
class Solution {
    static int[][] dic = {{0,1},{0,-1},{1,0},{-1,0}};
    public boolean exist(char[][] board, String word) {
        boolean[][] visited = new boolean[board.length][board[0].length];

        for(int i = 0; i < board.length; i++) {
            for(int j =0; j < board[0].length; j++) {
                if(word.charAt(0) == board[i][j]) {
                    visited[i][j] = true;
                    if(dfs(board, word, i, j, 1, visited)) return true;
                    visited[i][j] = false;
                }
            }
        }

        return false;
    }

    public boolean dfs(char[][] board, String word, int x,int y, int n, boolean[][] visited) {
        if(n >= word.length()) return true;
        for(int i = 0; i< 4; i++) {
            int nextX = x + dic[i][0];
            int nextY = y + dic[i][1];

            if(nextX < 0 || nextX >= board.length) continue;
            if(nextY < 0 || nextY >= board[0].length) continue;
            if(board[nextX][nextY]==word.charAt(n) && visited[nextX][nextY] == false) {
                visited[nextX][nextY] = true;
                if(dfs(board, word, nextX, nextY, n+1,visited)) return true;
                visited[nextX][nextY] = false;
            }
        }

        return false;
    }
}
```
