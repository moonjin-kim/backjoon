## CH 2685. Count the Number of Complete Components

```java

class Solution {
    static int[] factor = new int[50];
    static boolean[] visited;
    public int countCompleteComponents(int n, int[][] edges) {
        ArrayList<Integer>[] graph = new ArrayList[n];
        visited = new boolean[n];
        int answer = 0;
        init();

        for (int i = 0; i < n; i++) graph[i] = new ArrayList<>();
		
		for(int[] edge : edges) {
			graph[edge[0]].add(edge[1]);
			graph[edge[1]].add(edge[0]);
		}

        for(int i = 0; i < n; i++) {
            if(!visited[i]) {
                if(bfs(i, graph)) {
                    answer++;
                }
            }
        }

        return answer;
    }

    public boolean bfs(int root, ArrayList<Integer>[] graph) {
        System.out.println("root: " + root);
        Set<Integer> set = new HashSet<>();
        int edge = 0;

        Queue<Integer> queue = new LinkedList();
        queue.add(root);

        while(!queue.isEmpty()) {
            int cur = queue.poll();
            if(visited[cur]) {
                continue;
            }
            set.add(cur);
            visited[cur] = true;
            for(int node: graph[cur]) {
                if(visited[node]) {
                    continue;
                }
                edge++;
                queue.add(node);
            }
        }

        return edge == factor[set.size()-1];
    }

    public void init() {
        factor[1] = 1;
        for(int i = 1; i < 50; i++) {
            factor[i] = factor[i-1] + i;
        }
    }
}
```
