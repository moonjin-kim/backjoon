## 199. Binary Tree Right Side View

```java
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public List<Integer> rightSideView(TreeNode root) {
        ArrayList<Integer> answer = new ArrayList();
        Queue<Node> queue = new LinkedList();
        int dept = 0;

        if(root != null) {
            queue.add(new Node(root, 1));
        }

        while(!queue.isEmpty()) {
            Node cur = queue.poll();
            TreeNode curNode = cur.getTree();
            int curDept = cur.getDept();
            if(curDept > dept) {
                answer.add(curNode.val);
                dept = curDept;
            }

            if(curNode.right != null) {
                queue.add(new Node(curNode.right, curDept + 1));
            }
            if(curNode.left != null) {
                queue.add(new Node(curNode.left, curDept + 1));
            }
        }
            
        return answer;
    }
}

public class Node {
    TreeNode tree;
    int dept;

    public Node(TreeNode tree, int dept) {
        this.tree = tree;
        this.dept = dept;
    }

    public TreeNode getTree() {
        return tree;
    }

    public int getDept() {
        return dept;
    }
}
```
