// 22. Generate Parentheses
```java
class Solution {
    ArrayList<String> answer = new ArrayList();

    public List<String> generateParenthesis(int n) {
        generate(0,0,"",n);

        return answer;
    }

    public void generate(int dept, int sum, String parenthesis, int n) {
        if(dept/n == 2 && dept%2 == 0) {
            if(sum == 0) {
                answer.add(parenthesis);
            }
            return;
        }

        if(sum == 0) {
            generate(dept+1, 1, parenthesis + '(', n);
        } else {
            generate(dept+1, sum + 1, parenthesis + '(', n);
            generate(dept+1, sum - 1, parenthesis + ')', n);
        }
    }
}
```
