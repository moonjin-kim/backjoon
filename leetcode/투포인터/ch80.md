## ch80. 정렬된 배열에 중복 제거2

```java
class Solution {
    public int removeDuplicates(int[] nums) {
        HashMap<Integer, Integer> map = new HashMap();
        int start = 0;

        for(int i = 0; i < nums.length; i++) {
            map.put(nums[i], map.getOrDefault(nums[i], 0) + 1);
            int count = map.get(nums[i]);
            if(count > 2) {
                continue;
            }
            nums[start] = nums[i];
            start++;
        }

        return start;
    }
}
```
