345. Reverse Vowels of a String
```java
class Solution {
    String gather = "aeiou";
    public String reverseVowels(String s) {
        char[] arr = s.toCharArray();   
        int start = 0;
        int end = s.length() - 1;

        while(start < end) {
            char startA = Character.toLowerCase(arr[start]);
            if(gather.indexOf(startA)<0) {
                start++;
                continue;
            }
            while(start < end) {
                char endA = Character.toLowerCase(arr[end]);
                if(gather.indexOf(endA)>=0) {
                    char temp = arr[end];
                    arr[end] = arr[start];
                    arr[start] = temp;
                    end--;
                    break;
                }
                end--;
            }
            start++;
        }
        return String.valueOf(arr);
    }
}
```
