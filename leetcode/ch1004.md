class Solution {
    public int longestOnes(int[] nums, int k) {
        int max = 0;
        int sum = 0;
        int start = 0;
        int end = 0;
        while(end < nums.length) {
            sum += nums[end++];
            if(sum + k > max) {
                max = Math.max(max, end-start);
            } else {
                sum -= nums[start];
                start++;
            }
        }

        return max;
    }
}
