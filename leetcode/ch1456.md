class Solution {
    String given = "aeiou";
    public int maxVowels(String s, int k) {
        int max = 0;
        int cur = 0;
        for(int i =0; i<k; i++) {
            if (given.indexOf(s.charAt(i)) != -1) {
                cur++;
            }
        }
        max = cur;
        for(int i=1; i<s.length()-k+1;i++) {
            if (given.indexOf(s.charAt(i-1)) != -1) {
                cur--;
            }
            if (given.indexOf(s.charAt(i+k-1)) != -1) {
                cur++;
            }
            max = Math.max(cur, max);
        }

        
        return max;
    }
}
