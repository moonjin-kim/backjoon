class Solution {
    public int maxProfit(int[] prices) {
        int max = 0;
        int min = prices[0];

        // n
        for(int i = 1; i < prices.length; i++) {
            if(min > prices[i]) {
                min = prices[i];
            }
            max = Math.max(prices[i] - min, max);
        }

        return max;
    }
}
