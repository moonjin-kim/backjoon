## ch875 : 바나마를 먹는 코코

```java
class Solution {
    public int minEatingSpeed(int[] piles, int h) {
        Arrays.sort(piles);
        int start = 1;
        int end = piles[piles.length-1];
        int answer = end;

        while(start <= end) {
            int mid = start + (end - start) / 2;
            boolean isFinish = canFinish(piles, mid, h);
            
            if(isFinish) {
                answer = mid;
                end = mid - 1;
            } else {
                start = mid + 1;
            }
        }

        return answer;
    }

    public boolean canFinish(int[] piles, int k, int h) {
        long sum = 0;
        for(int pile : piles) {
            sum += pile / k;
            if(pile % k != 0) {
                sum++;
            }
        }

        return sum <= h;
    }
}
```
