## 77. Combinations

```java
class Solution {
    static List<List<Integer>> perm;
    public List<List<Integer>> combine(int n, int k) {
        perm = new ArrayList();

        dfs(new ArrayList(), 1, n, 0, k);

        return perm;
    }

    public void dfs(List<Integer> list, int num, int maxLen, int dept, int max) {
        if(dept == max) {
            perm.add(new ArrayList<>(list));
            return;
        }

        for(int i = num; i <= maxLen; i++) {
            list.add(i);
            dfs(list, i+1, maxLen, dept + 1, max);
            list.remove(list.size() - 1);
        }
    }
}
```
