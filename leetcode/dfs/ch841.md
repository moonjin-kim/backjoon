## 841. Keys and Rooms

```java
class Solution {
    public boolean canVisitAllRooms(List<List<Integer>> rooms) {
        boolean[] visit = new boolean[rooms.size()];
        visit[0] = true;

        dfs(rooms, 0, visit);
         for(int i = 0; i < rooms.size(); i++) {
            if(!visit[i]) return false;
         }

         return true;
    }

    public void dfs(List<List<Integer>> rooms, int roomNum, boolean[] visit) {
        List<Integer> room = rooms.get(roomNum);

        for(int key : room) {
            if(!visit[key]) {
                visit[key] = true;
                dfs(rooms, key, visit);
            }
        }
    }
}
```
