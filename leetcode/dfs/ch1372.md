/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public int longestZigZag(TreeNode root) {
        int max = 0;
        int right = zigZag(root.right, 1, 0);
        int left = zigZag(root.left, 1, 1);
        max = Math.max(left, max);
        max = Math.max(right, max);

        return max;
    }

    public int zigZag(TreeNode node, int len, int dir) {
        int max = len;
        if(node == null) return len - 1;
        if(dir==1) {
            int right = zigZag(node.right, len + 1, 0);
            int left = zigZag(node.left, 1, 1);
            max = Math.max(left, max);
            max = Math.max(right, max);
        } else {
            int right = zigZag(node.right, 1, 0);
            int left = zigZag(node.left, len+ 1, 1);
            max = Math.max(left, max);
            max = Math.max(right, max);
        }

        return max;
    }
}
