## 순열

```java
class Solution {
    static List<List<Integer>> perm;
    public List<List<Integer>> permuteUnique(int[] nums) {
        perm = new ArrayList();
        boolean[] visited = new boolean[nums.length];
        Arrays.sort(nums);
        dfs(nums, new ArrayList(), 0, visited);

        return perm;
    }

    public void dfs(int[] nums, List<Integer> list, int dept, boolean[] visited) {
        if(dept == nums.length) {
            perm.add(new ArrayList<>(list));
            return;
        }

        for(int i = 0; i < nums.length; i++) {
            if (visited[i]) continue;

            // 중복 제거 핵심:
            // nums[i] == nums[i-1] 인데 i-1이 아직 사용되지 않았다면,
            // 같은 depth에서 nums[i]를 먼저 쓰는 케이스는 중복을 만든다.
            if (i > 0 && nums[i] == nums[i - 1] && !visited[i - 1]) continue;

            list.add(nums[i]);
            visited[i] = true;
            dfs(nums, list, dept + 1, visited);
            list.remove(list.size() - 1);
            visited[i] = false;
        }
    }
}
```
