17. Letter Combinations of a Phone Number
```java
class Solution {
    Map<Character, String> map = new HashMap();
    ArrayList<String> answer = new ArrayList();

    public List<String> letterCombinations(String digits) {
        if(digits.length() == 0) {
            return new ArrayList();
        }
        init();
        cons(0, digits, "");

        return answer;
    }

    public void cons(int dept, String digits, String data) {
        if(dept == digits.length()) {
            answer.add(data);
            return;
        }
        Character number = digits.charAt(dept);
        String letter = map.get(number);
        for(int i = 0; i < letter.length(); i++) {
            Character word = letter.charAt(i);
            cons(dept + 1, digits, data + word);
        }
    }

    public void init() {
        map.put('2', "abc");
        map.put('3', "def");
        map.put('4', "ghi");
        map.put('5', "jkl");
        map.put('6', "mno");
        map.put('7', "pqrs");
        map.put('8', "tuv");
        map.put('9', "wxyz");
    }
}
```
