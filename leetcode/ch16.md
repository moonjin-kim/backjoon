3Sum Closest

```java
class Solution {
    public int threeSumClosest(int[] nums, int target) {
        int min = 1000000;
        int answer = 0;

        Arrays.sort(nums);
        for(int i=0; i<nums.length - 2; i++) {
            int start = i + 1;
            int end = nums.length - 1;
            while(start < end) {
                int sum = nums[i] + nums[start] + nums[end];
                if(sum < target) {
                    start++;
                } else if(sum > target) {
                    end--;
                } else {
                    return sum;
                }
                
                if(min > Math.abs(sum - target)) {
                    min = Math.abs(sum - target);
                    answer = sum;
                }
            }
        }

        return answer;
    }

}
```
