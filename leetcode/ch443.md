## 443. String Compression

```java
class Solution {
    public int compress(char[] chars) {
        String str = "";
        char cur = chars[0];
        int count = 1;
        str+=cur;
        for(int i = 1; i < chars.length; i++) {
            if(cur == chars[i]) count++;
            else {
                if(count >1) {
                    str+=count;
                }
                cur = chars[i];
                str+=cur;
                count = 1;
            }
        }
        if(count >1) {
            str+=count;
        }

        for (int i = 0; i < str.length(); i++) {
            chars[i] = str.charAt(i);
        }
        return str.length();
    }
}
```
