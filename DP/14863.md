## 백준 14563 서울에서 경산
```java
package com.backjoon.back14863;

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.LinkedList;
import java.util.Queue;
import java.util.Scanner;
import java.util.StringTokenizer;

public class Main {
    static int n;
    static int maxTime;
    static int answer;
    public static void main(String args[]) throws IOException {
        BufferedReader reader = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer tokenizer = new StringTokenizer(reader.readLine());

        n = Integer.parseInt(tokenizer.nextToken());
        maxTime = Integer.parseInt(tokenizer.nextToken());
        answer = 0;

        int[][][] cost = new int[n+1][2][2];
        for(int i = 1; i <= n; i++) {
            tokenizer = new StringTokenizer(reader.readLine());
            cost[i][0][0] = Integer.parseInt(tokenizer.nextToken());
            cost[i][0][1] = Integer.parseInt(tokenizer.nextToken());
            cost[i][1][0] = Integer.parseInt(tokenizer.nextToken());
            cost[i][1][1] = Integer.parseInt(tokenizer.nextToken());
        }

        int[][] dp = new int[n+1][maxTime+1];
        dp[1][cost[1][0][0]] = cost[1][0][1];
        dp[1][cost[1][1][0]] = Math.max(dp[1][cost[1][1][0]], cost[1][1][1]);

        for(int i = 1; i <= n; i++) {
            for(int j = maxTime ; j >= cost[i][0][0] ; j--) {
                if(dp[i-1][j - cost[i][0][0]] != 0) {
                    dp[i][j] = Math.max(dp[i][j] , dp[i-1][j - cost[i][0][0]] + cost[i][0][1]);
                }
            }

            for(int j = maxTime ; j >= cost[i][1][0] ; j--) {
                if(dp[i-1][j - cost[i][1][0]] != 0) {
                    dp[i][j] = Math.max(dp[i][j] , dp[i-1][j - cost[i][1][0]] + cost[i][1][1]);
                }
            }
        }

        int answer = 0;

        for(int i = 1; i <= maxTime; i++) {
            answer = Math.max(answer, dp[n][i]);
        }

        System.out.println(answer);
    }
}
```
