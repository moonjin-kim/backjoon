## 가장먼노드

```java
import java.util.*;

class Solution {
    int[] visited;
    int max = 0;
    int maxCount = 0;
    public int solution(int n, int[][] edges) {
        int answer = 0;
        LinkedList[] linkedList = new LinkedList[n+1];
        for(int i = 0; i < n+1; i++){
            linkedList[i] = new LinkedList<>();
        }
        
        for(int[] edg : edges) {
            linkedList[edg[0]].add(edg[1]);
            linkedList[edg[1]].add(edg[0]);
        }
        
        visited = new int[n+1];
        Arrays.fill(visited, 500001);
        visited[0] = 0;
        visited[1] = 0;
        
        bfs(edges,linkedList);
        
        return maxCount;
    }
    
    public void bfs(int[][] edges, LinkedList[] linkedList) {
        Queue<Integer> q = new LinkedList<Integer>();
        q.offer(1);
        while(!q.isEmpty()) {
            int currentIdx = q.poll();
            
            for(int i = 0; i < linkedList[currentIdx].size(); i++){
                int nextIdx = (int)linkedList[currentIdx].get(i);
                if(visited[nextIdx] > visited[currentIdx] + 1) {
                    visited[nextIdx] = visited[currentIdx] + 1;
                    q.offer(nextIdx);
                    
                    checkMaxDist( visited[currentIdx] + 1);
                }
            }
        }
    }
    
    public void checkMaxDist(int dist){
        if(dist > max) {
            max = dist;
            maxCount = 1;
        } else if(dist == max){
            maxCount++;
        }
    }
}
```
