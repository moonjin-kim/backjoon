## 부대복귀

```java
import java.util.*;

class Solution {
    public int[] solution(int n, int[][] roads, int[] sources, int destination) {
        int[] answer = new int[sources.length];
        Map<Integer, ArrayList<Integer>> map = initMap(n, roads);
        
        int[] visited = bfs(n, destination, map);
        for(int i = 0; i <sources.length; i++) {
            answer[i] = visited[sources[i]];
            if(answer[i] == Integer.MAX_VALUE) answer[i] = -1;
        }
        
        return answer;
    }
    
    public int[] bfs(int n, int source, Map<Integer, ArrayList<Integer>> map) {
        Queue<Integer> queue = new LinkedList();
        queue.add(source);
        
        int[] visited = new int[n+1];
        Arrays.fill(visited, Integer.MAX_VALUE);
        visited[source] = 0;
        
        while(!queue.isEmpty()) {
            int cur = queue.poll();
            ArrayList<Integer> roads = map.get(cur);
            
            for(int road: roads) {
                if(visited[road] > visited[cur] + 1) {
                    queue.add(road);
                    visited[road] = visited[cur] + 1;
                }
            }
        }
        
        return visited;
    }
    
    public Map<Integer, ArrayList<Integer>> initMap(int n, int[][] roads) {
        Map<Integer, ArrayList<Integer>> map = new HashMap();
        
        for(int i = 1; i <= n; i++) {
            map.put(i, new ArrayList());
        }
        
        for(int[] road: roads) {
            ArrayList<Integer> target1 = map.get(road[0]);
            ArrayList<Integer> target2 = map.get(road[1]);
            
            target1.add(road[1]);
            target2.add(road[0]);
        }
        
        return map;
    }
}
```
