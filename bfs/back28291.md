package com.backjoon.back28291;

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.LinkedList;
import java.util.Queue;
import java.util.StringTokenizer;

public class Main {
    static int[] dx = {0,0,1,-1};
    static int[] dy = {1,-1,0,0};
    public static void main(String[] args) throws IOException {
        BufferedReader reader = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer tokenizer = new StringTokenizer(reader.readLine());
        int W = Integer.parseInt(tokenizer.nextToken());
        int H = Integer.parseInt(tokenizer.nextToken());
        int n = Integer.parseInt(reader.readLine());
        int[][] board = new int[W][H];
        int[][] visited = new int[W][H];

        Queue<Electronic> q = new LinkedList<>();

        int count = 0;
        for (int i = 0; i < n; i++) {
            tokenizer = new StringTokenizer(reader.readLine());
            String blockName = tokenizer.nextToken();
            int x = Integer.parseInt(tokenizer.nextToken());
            int y = Integer.parseInt(tokenizer.nextToken());
            if(blockName.equals("redstone_block")){
                board[x][y] = 2;
                q.add(new Electronic(x, y, 16));
                visited[x][y] = 16;
            } else if(blockName.equals("redstone_dust")){
                board[x][y] = 1;
            } else if(blockName.equals("redstone_lamp")){
                board[x][y] = 3;
                count++;
            }
        }

        boolean isSuccess = solve(board, q, visited, count);
        if(isSuccess){
            System.out.println("success");
        } else{
            System.out.println("failed");
        }
    }

    public static boolean solve(int[][] board, Queue<Electronic> q, int[][] visited, int count) {
        while(!q.isEmpty()){
            Electronic e = q.poll();

            for(int i=0;i<4;i++){
                int nx = e.x + dx[i];
                int ny = e.y + dy[i];

                if(!isMove(nx,ny, visited)) continue;
                if(board[nx][ny] == 1 && visited[nx][ny] < e.energy - 1) {
                    visited[nx][ny] = e.energy - 1;
                    q.offer(new Electronic(nx,ny,e.energy - 1));
                } else if(board[nx][ny] == 3 && 0 < e.energy - 1 && visited[nx][ny] == 0) {
                    count--;
                    visited[nx][ny] = 20;
                }
            }
        }

        return count == 0;
    }

    public static boolean isMove(int nx, int ny, int[][] visited) {
        return nx >= 0 && nx < visited.length && ny >= 0 && ny < visited[0].length;
    }

    public static class Electronic {
        public int x;
        public int y;
        public int energy;

        public Electronic(int x, int y, int energy) {
            this.x = x;
            this.y = y;
            this.energy = energy;
        }
    }
}
