package com.backjoon.back30894;

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.ArrayList;
import java.util.LinkedList;
import java.util.Queue;
import java.util.StringTokenizer;

public class Main {
    static int[] dx = {0, 1, 0,-1,0};
    static int[] dy = {1, 0, -1, 0,0};
    public static void main(String[] args) throws IOException {
        BufferedReader reader = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer tokenizer = new StringTokenizer(reader.readLine());

        int N = Integer.parseInt(tokenizer.nextToken());
        int M = Integer.parseInt(tokenizer.nextToken());
        char[][][] map = new char[N][M][4];

        tokenizer = new StringTokenizer(reader.readLine());
        int startX = Integer.parseInt(tokenizer.nextToken());
        int startY = Integer.parseInt(tokenizer.nextToken());
        int endX = Integer.parseInt(tokenizer.nextToken());
        int endY = Integer.parseInt(tokenizer.nextToken());

        for(int i = 0; i < N; i++){
            tokenizer = new StringTokenizer(reader.readLine());
            String command = tokenizer.nextToken();
            for(int j = 0; j < M; j++){
                map[i][j][0] = command.charAt(j);
                map[i][j][1] = command.charAt(j);
                map[i][j][2] = command.charAt(j);
                map[i][j][3] = command.charAt(j);
            }
        }

        int[][][] visited = ghostMap(map);

        int count = bfs(map, visited, startX, startY, endX, endY);

        if(count == -1) {
            System.out.println("GG");
        } else {
            System.out.println(count);
        }
    }

    // 각 시간별 유령이 보는 방향 기록
    public static int[][][] ghostMap(char[][][] map){
        int[][][] visited = new int[map.length][map[0].length][4];
        ArrayList<int[]> list = new ArrayList<>();
        for(int i = 0; i < map.length; i++){
            for(int j = 0; j < map[0].length; j++){
                if(map[i][j][0] == '0' || map[i][j][0] == '1' || map[i][j][0] == '2'||map[i][j][0] == '3'){
                    list.add(new int[]{i, j});
                }
            }
        }

        for(int i =0; i < 4; i++){
            for(int[] ghost : list){
                int dic = (map[ghost[0]][ghost[1]][i] + i) % 4;
                int nextX = ghost[0];
                int nextY = ghost[1];

                while(true) {
                    nextX = nextX + dx[dic];
                    nextY = nextY + dy[dic];

                    if(nextX < 0 || nextX >= map.length || nextY < 0 || nextY >= map[0].length) break;
                    if(map[nextX][nextY][i] != '.') break;
                    visited[nextX][nextY][i] = 100000;
                }
            }
        }

        return visited;
    }

    public static int bfs(char[][][] map, int[][][] visited,int startX, int startY, int endX, int endY){
        Queue<int[]> queue = new LinkedList<>();
        queue.add(new int[]{startX-1, startY-1, 0});
        visited[startX-1][startY-1][0] = 1;

        if(map[startX-1][startY-1][0] == 5) return -1;

        while(!queue.isEmpty()){
            int[] cur = queue.poll();
            if(cur[0] == endX-1 && cur[1] == endY-1)
                return cur[2];

            if(map[cur[0]][cur[1]][cur[2] % 4] == '5') continue;

            for(int i = 0; i < 5; i++){
                int nextX = cur[0] + dx[i];
                int nextY = cur[1] + dy[i];
                int dic = (cur[2] + 1) % 4;
                if(nextX < 0 || nextX >= map.length || nextY < 0 || nextY >= map[0].length) continue;
                if(visited[nextX][nextY][dic] > 0) continue;
                if(map[nextX][nextY][dic] == '#') continue;
                if(map[nextX][nextY][dic] <= '5' && map[nextX][nextY][dic] >= '0') continue;

                queue.add(new int[]{nextX, nextY, cur[2]+1});
                visited[nextX][nextY][dic] = cur[2] + 1;
            }
        }

        return -1;
    }
}
