## 게임 맵 최단거리

```java
import java.util.*;

class Solution {
    int[][] dic = {{1,0},{-1,0},{0,1},{0,-1}};
    public int solution(int[][] maps) {
        Queue<Node> queue = new LinkedList<>();
        int n = maps.length;
        int m = maps[0].length;
        int answer = 0;
        int[][] visited = new int[n][m];
        queue.add(new Node(0,0));
        visited[0][0] = 1;
        
        while(!queue.isEmpty()) {
            Node cur = queue.poll();
            
            for(int i=0; i<4; i++) {
                int nextX = cur.x + dic[i][0];
                int nextY = cur.y + dic[i][1];
                if(nextX < 0 || nextX >= n) continue;
                if(nextY < 0 || nextY >= m) continue;
                if(maps[cur.x][cur.y] == 0) continue;
                
                if(visited[nextX][nextY] == 0) {
                    visited[nextX][nextY] = visited[cur.x][cur.y] + 1;
                    queue.add(new Node(nextX, nextY));
                }
            }
        }
        
        if(visited[n-1][m-1] == 0) return -1;
        return visited[n-1][m-1];
    }
}

class Node {
    public int x;
    public int y;
    
    public Node(int x, int y) {
        this.x = x;
        this.y = y;
    }
}
```
