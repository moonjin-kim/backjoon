package com.backjoon.back2473;

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.Arrays;
import java.util.StringTokenizer;

public class Main {
    public static void main(String[] args) throws IOException {
        BufferedReader reader = new BufferedReader(new InputStreamReader(System.in));

        int n = Integer.parseInt(reader.readLine());
        long[] nums = new long[n];
        StringTokenizer tokenizer = new StringTokenizer(reader.readLine());
        for (int i = 0; i < n; i++) {
            nums[i] = Long.parseLong(tokenizer.nextToken());
        }
        long[] result = solve(nums);
        for(int i = 0; i< 3; i++) {
            System.out.print(result[i] + " ");
        }
    }

    public static long[] solve(long[] nums) {
        long min = Long.MAX_VALUE;
        long[] minTarget = new long[3];
        Arrays.sort(nums);

        for (int i = 0; i < nums.length-2; i++) {
            int start = i + 1;
            int end = nums.length-1;

            while(start < end) {
                long num = nums[i] + nums[start] + nums[end];
                if(Math.abs(num) < min) {
                    min = Math.abs(num);
                    minTarget[0] = nums[i];
                    minTarget[1] = nums[start];
                    minTarget[2] = nums[end];
                }

                if(num > 0) end--;
                else if(num < 0) start++;
                else return new long[] {nums[i], nums[start], nums[end]};
            }
        }

        return minTarget;
    }
}
