package com.backjoon.back10655;

import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.util.StringTokenizer;

public class Main {
    public static void main(String[] args) throws Exception {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        int n = Integer.parseInt(br.readLine());
        int[][] arr = new int[n][2];
        int[][] dp = new int[n][2];

        for (int i = 0; i < n; i++) {
            StringTokenizer st = new StringTokenizer(br.readLine());
            arr[i][0] = Integer.parseInt(st.nextToken());
            arr[i][1] = Integer.parseInt(st.nextToken());
        }

        dp[0][0] = 0;
        dp[1][0] = Math.abs(arr[0][0] - arr[1][0]) + Math.abs(arr[0][1] - arr[1][1]);
        dp[2][0] = dp[1][0] + Math.abs(arr[1][0] - arr[2][0]) + Math.abs(arr[1][1] - arr[2][1]);
        dp[2][1] = Math.abs(arr[0][0] - arr[2][0]) + Math.abs(arr[0][1] - arr[2][1]);
        for(int i = 3; i < n; i++) {
            dp[i][0] = dp[i-1][0] + Math.abs(arr[i-1][0] - arr[i][0]) + Math.abs(arr[i-1][1] - arr[i][1]);
            int a = dp[i-2][0] + Math.abs(arr[i-2][0] - arr[i][0]) + Math.abs(arr[i-2][1] - arr[i][1]);
            int b = dp[i-1][1] + Math.abs(arr[i-1][0] - arr[i][0]) + Math.abs(arr[i-1][1] - arr[i][1]);
            dp[i][1] = Math.min(a, b);
        }

        System.out.println(dp[n-1][1]);
    }
}
