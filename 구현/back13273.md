## 13273 로마숫자

```java
package com.younghan.back13273;

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.HashMap;
import java.util.Map;

public class back13273 {
    static HashMap<Integer, String[]> romeMap = new HashMap<>();
    public static void main(String[] args) throws IOException {
        BufferedReader reader = new BufferedReader(new InputStreamReader(System.in));
        int n = Integer.parseInt(reader.readLine());

        for(int i = 0; i < n; i++){
            String num = reader.readLine();
            System.out.println(solve(num));
        }
    }

    public static String solve(String s) {
        makeMap();
        if(isInteger(s)) {
            return intToRome(Integer.parseInt(s));
        } else {
            return RomeToInt(s);
        }
    }

    public static void makeMap() {
        String[] one = new String[]{"","I","II","III","IV","V","VI","VII","VIII","IX"};
        String[] ten = new String[]{"","X","XX","XXX","XL","L","LX","LXX","LXXX","XC"};
        String[] hun = new String[]{"","C","CC","CCC","CD","D","DC","DCC","DCCC","CM"};
        String[] tho = new String[]{"","M", "MM", "MMM"};

        romeMap.put(1, one);
        romeMap.put(2, ten);
        romeMap.put(3, hun);
        romeMap.put(4, tho);
    }

    public static String intToRome(int num) {
        String answer = "";
        int[] div = new int[]{0,1,10,100,1000};
        for(int i =4; i > 0; i--) {
            String[] table = romeMap.get(i);
            int index = num / div[i];

            answer = answer + table[index];
            num = num % div[i];
        }

        return answer;
    }

    public static String RomeToInt(String rome) {
        Map<Character, Integer> map = new HashMap<>();
        map.put('I', 1);
        map.put('V', 5);
        map.put('X', 10);
        map.put('L', 50);
        map.put('C', 100);
        map.put('D', 500);
        map.put('M', 1000);

        int total = 0;

        for (int i = 0; i < rome.length(); i++) {
            char c = rome.charAt(i);

            int cur = map.get(c);
            int next = 0;

            if (i + 1 < rome.length()) {
                next = map.get(rome.charAt(i + 1));
            }

            if(next > cur) {
                total -= cur;
            } else {
                total += cur;
            }
        }

        return "" + total;
    }

    public static boolean isInteger(String strValue) {
        try {
            Integer.parseInt(strValue);
            return true;
        } catch (NumberFormatException ex) {
            return false;
        }
    }
}
```
