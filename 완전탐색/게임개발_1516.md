package com.backjoon.back1516;

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.*;

public class Main {
    public static void main(String[] args) throws IOException {
        BufferedReader reader = new BufferedReader(new InputStreamReader(System.in));

        int n = Integer.parseInt(reader.readLine());
        int[] buildTimes = new int[n+1];
        int[] indegree = new int[n+1];
        Map<Integer, List<Integer>> map = new HashMap<>();
        for(int i = 1; i <= n; i++) {
            map.put(i, new ArrayList<>());
        }

        for(int i = 1; i <= n; i++) {
            StringTokenizer tokenizer = new StringTokenizer(reader.readLine());
            buildTimes[i] = Integer.parseInt(tokenizer.nextToken());
            while(true) {
                int key = Integer.parseInt(tokenizer.nextToken());
                if(key == -1) break;

                // i번째 건물이 직접적으로 필요한 건물 개수
                indegree[i]++;
                // key 건물을 건설하였을때 지을 수 있는 건물 리스트
                map.get(key).add(i);
            }
        }


        int[] result = solve(n, indegree, buildTimes, map);
        for(int i = 1; i <= n; i++) {
            System.out.println(result[i]);
        }
    }

    public static int[] solve(int n, int[] indegree, int[] buildTimes,Map<Integer, List<Integer>> buildMap) {
        int[] result = new int[n+1];

        Queue<Integer> que = new LinkedList<>();

        // 필요한 건물이 없는 건물로 시작
        for(int i = 1; i <= n; i++) {
            if(indegree[i] == 0) {
                que.add(i);
            }
        }


        while(!que.isEmpty()) {
            int curr = que.poll();
            List<Integer> build = buildMap.get(curr);

            for(int i = 0; i < build.size(); i++) {
                int next = build.get(i);
                indegree[next]--;
                result[next] = Math.max(result[next], result[curr] + buildTimes[curr]);
                // 선행 건물이 모두 지어졌을때
                if(indegree[next] == 0) {
                    que.add(next);
                }
            }
        }

        for(int i = 1; i <= n; i++) {
            result[i] = result[i] + buildTimes[i];
        }

        return result;
    }

}
