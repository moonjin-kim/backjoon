
### 완전탐색 풀이
```java
// 지속해서 전체 국회의원의 투표수를 조회하여 반복횟수마다 가장 점수자의 표를 0번에 준다
package com.younghan.back1417;

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.PriorityQueue;
import java.util.StringTokenizer;

public class back1417 {

    public static void main(String[] args) throws IOException {
        BufferedReader reader = new BufferedReader(new InputStreamReader(System.in));

        int n = Integer.parseInt(reader.readLine());
        int[] arr = new int[n];
        StringTokenizer tokenizer = new StringTokenizer(reader.readLine());
        for(int i = 0; i < n; i++){
            arr[i] = Integer.parseInt(tokenizer.nextToken());
        }

        System.out.println(solveBacktraking(n,arr));
    }

    public static int solveBacktraking(int n, int[] arr) {
        int count = 0;

        while(true) {
            int max = arr[0];
            int maxNum = 0;
            for(int i = 1; i < n; i++) {
                if(arr[i] >= max) {
                    max = arr[i];
                    maxNum = i;
                }
            }

            if(maxNum == 0) break;
            arr[maxNum] -= 1;
            arr[0] += 1;
            count++;
        }

        return count;
    }
}

```

### 우선순위 큐 풀이
```java

// 현재 0번을 제외한 후보자의 포를 우선순위 큐에 넣으면 가장 상당이 최다 득표자임으로 현재 최다득표자와 기호0번의 득표자와 비교해서 표를 전달한다

    public static int solve(int n, int[] arr) {
        //현재 투표수 중에서 가장 큰 값을 큐 상단으로 패치
        PriorityQueue<Node> pq = new PriorityQueue<>((a, b) -> b.vote - a.vote);
        int count = 0;
        int curVote = arr[0];
        if(arr.length == 1) return 0;

        for(int i = 1; i < n; i++) {
            pq.offer(new Node(i, arr[i]));
        }
        System.out.println();
        while(!pq.isEmpty()) {
            Node maxVote = pq.poll();

            if(maxVote.vote < curVote) break;
            curVote += 1;
            count++;
            pq.offer(new Node(maxVote.num, maxVote.vote - 1));
        }

        return count;
    }

    static class Node {
        public int num;
        public int vote;

        public Node(int num, int vote) {
            this.num = num;
            this.vote = vote;
        }
    }
```
