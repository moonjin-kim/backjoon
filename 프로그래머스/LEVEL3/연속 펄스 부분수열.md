class Solution {
    public long solution(int[] sequence) {
        long purse1 = 0;        
        long purse2 = 0;
        long answer = 0;
        int step1 = 1;
        int step2 = -1;
        
        for(int i = 0; i<sequence.length; i++) {
            purse1 += sequence[i] * step1;
            purse2 += sequence[i] * step2;
            
            purse1 = Math.max(0, purse1);
            purse2 = Math.max(0, purse2);
            
            answer = Math.max(answer, Math.max(purse1, purse2));
            
            step1 = step1 * -1;
            step2 = step2 * -1;
        }
        
        
        return answer;
    }
}
