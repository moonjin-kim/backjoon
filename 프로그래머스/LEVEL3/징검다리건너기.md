class Solution {
    public int solution(int[] stones, int k) {
        int n = stones.length;
        int loStone = Integer.MAX_VALUE;
        int hiStone = Integer.MIN_VALUE;
        for(int i = 0; i < stones.length; i++) {
            loStone = Math.min(loStone, stones[i]);
            hiStone = Math.max(hiStone, stones[i]);
        }
        
        while(loStone < hiStone) {
            int mid = (loStone + hiStone) / 2;
            
            if(canJump(stones, k, mid)) {
                loStone = mid + 1;
            } else {
                hiStone = mid;
            }
        }
        
        return loStone;
    }
    
    boolean canJump(int[] stones, int k, int mid) {
        int cnt = 0;
        for (int i = 0; i < stones.length; i++) {
            if(stones[i] - mid <= 0) cnt++;
            else cnt = 0;
            
            if(cnt == k) return false;
        }
        
        return true;
    }
}
