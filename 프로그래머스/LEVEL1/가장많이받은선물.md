## 프로그래머스 가장많이받은선물

```java
import java.util.*;

class Solution {
    
    public int solution(String[] friends, String[] gifts) {
        int n = friends.length;
        Map<String, Integer> nameMap = new HashMap();
        for(int i = 0; i < friends.length; i++) {
            nameMap.put(friends[i],i);
        }
        
        int[] giftPoint = new int[n];
        int[][] points = new int[n][n];
        for(String gift: gifts) {
            int sender = nameMap.get(gift.split(" ")[0]);
            int receiver = nameMap.get(gift.split(" ")[1]);
            points[sender][receiver]  += 1;
            points[receiver][sender]  -= 1;
            giftPoint[sender] += 1;
            giftPoint[receiver] -= 1;
        }
        
        int[] nextGift = new int[n];
        for(int i = 0; i < n; i++) {
            for(int j = i+1; j < n; j++) {
                int iPoint = giftPoint[i];
                int jPoint = giftPoint[j];
                
                if(points[i][j] > 0) nextGift[i]++;
                else if(points[i][j] < 0) nextGift[j]++;
                else if(points[i][j] == 0 && iPoint > jPoint) nextGift[i]++;
                else if(points[i][j] == 0 && iPoint < jPoint) nextGift[j]++;
            }
        }
        
        int answer = 0;
        for(int i = 0; i < n; i++) {
            answer = Math.max(answer, nextGift[i]);
        }
        
        return answer;
    }
}
```
