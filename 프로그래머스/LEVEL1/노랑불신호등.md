class Solution {
    public int solution(int[][] signals) {
        int answer = 0;
        
        int len = 1;
        
        // 전체 사이클 계산
        for (int i=0; i<signals.length; i++) {
            int g = signals[i][0];
            int y = signals[i][1];
            int r = signals[i][2];
            
            len *= (g+y+r);
        }
        
        int[] arr = new int[len+1];
        for(int i = 0; i<signals.length; i++) {
            int g = signals[i][0];
            int y = signals[i][1];
            int r = signals[i][2];
            
            int offset = (g + y + r);
            // j : 초록불이 켜진 시간
            for(int j = 1; j <= len; j+=offset) {
                
                for(int k = 0; k < y; k++) {
                    // 해당 시간에 노란불이 켜진 개수
                    arr[g+j+k]++;
                }
            }
        }
        
        
        for(int i = 1; i <= len; i++) {
            //해당 시간에 켜진 횟수가 총 신호등 개수일때
            if(arr[i] == signals.length) {
                return i;
            }
        }
        
        return -1;
    }
}
