class Solution {
    public int solution(int[] schedules, int[][] timelogs, int startday) {
        int answer = 0;
        
        for(int i = 0; i < schedules.length; i++) {
            schedules[i] = calc(schedules[i]) + 10;
            int count = 0;
            int day = startday;
            for(int j = 0; j < 7; j++) {
                if(schedules[i] >= calc(timelogs[i][j]) && day != 6 && day != 7) count++;
                day = day % 7 + 1;
            }
            
            if(count == 5) answer++;
        }
        
        return answer;
    }
    
    int calc(int time) {
        int hour = time / 100;
        int min = time % 100;
        return hour*60 + min;
    }
}
