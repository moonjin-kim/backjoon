## 카카오 2026 1차 중요한 단어들

```java
import java.util.*;

class Solution {
    public int solution(String message, int[][] spoiler_ranges) {
        int answer = 0;
        HashSet<String> importantWord = new HashSet();
        boolean[] isSpoiled = makeSpoiler(message, spoiler_ranges,importantWord);
        
        int i = 0;
        while(i < message.length()) {
            if (message.charAt(i) == ' ') {
                i++;
                continue;
            }
            
            int start = i;
            while(i < message.length() && message.charAt(i) != ' ') {
                i++;
            }
            
            int end = i;
            if (!isSpoiled[start]) {
                importantWord.remove(message.substring(start,end));
            } 
        }
        
        
        return importantWord.size();
    }
    
    public boolean[] makeSpoiler(
        String message, 
        int[][] spoiler_ranges,
        HashSet<String> importantWord
    ) {
        boolean[] isSpoiler = new boolean[message.length()];
        
        for(int[] spoiler: spoiler_ranges) {
            int start = spoiler[0];
            int end = spoiler[1];
            start = moveStart(message, start);
            end = moveEnd(message, end);
            for(int i = start; i <= end; i++) {
                isSpoiler[i] = true;
            }
            for(String word: message.substring(start,end+1).trim().split(" ")) {
                if(word.equals("")) continue;
                importantWord.add(word);
            }
        }
        
        return isSpoiler;
    }
    
    public int moveStart(String message, int start) {
        if(message.charAt(start) == ' ') return start;
        while(start > 0 && message.charAt(start-1) != ' ') {
            start--;
        }
        
        return start;
    }
    
    public int moveEnd(String message, int end) {
        if(message.charAt(end) == ' ') return end;
        while(end < message.length() - 1 && message.charAt(end+1) != ' ') {
            end++;
        }
        
        return end;
    }
}
```
